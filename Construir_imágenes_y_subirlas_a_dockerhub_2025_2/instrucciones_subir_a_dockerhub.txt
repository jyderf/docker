# Crear imagen desde contenedor: en este caso el contenedor debe existir y puede estar corriendo o detenido
docker commit ser1 miusuario/mi_imagen1:1.0

# Verificar que la imagen existe
docker images

# Login a Docker Hub
docker login

# Subir la imagen a tu repositorio
docker push miusuario/mi_imagen2:1.0
